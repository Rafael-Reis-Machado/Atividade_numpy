''' Exercício 4: Criar um objeto do tipo Cadastro e carregar nele o conteúdo
do arquivo teste.txt que foi criado no Exercício 3. Na sequência,
imprimir na tela o conteúdo do objeto Cadastro. '''

from aula.codigo import *

a = Cliente('Ana Maria', 25, 1)
b = Cliente('Pedro Silva', 21, 2)
c = Cliente('Amanda Soares', 28, 3)

d = Cadastro()
d.add(a)
d.add(b)
d.add(c)

d.salvar('teste.txt') #com o mÉtodo SALVAR colocamos o conteudo no TXT

e = Cadastro() 
e.ler('teste.txt') #com o método LER pegamos o conteudo que foi salvo no TXT
e.print()
